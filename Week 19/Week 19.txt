// ── Doubly Linked List ───────────────────────────────────────
class DLLNode {
  constructor(song, artist, emoji) {
    this.song   = song;
    this.artist = artist;
    this.emoji  = emoji;
    this.prev   = null;
    this.next   = null;
  }
}

class DoublyLinkedList {
  constructor() {
    this.head = null;
    this.tail = null;
    this.size = 0;
  }

  addFront(song, artist, emoji) {
    const node = new DLLNode(song, artist, emoji);
    if (!this.head) {
      this.head = this.tail = node;
    } else {
      node.next       = this.head;
      this.head.prev  = node;
      this.head       = node;
    }
    this.size++;
  }

  addEnd(song, artist, emoji) {
    const node = new DLLNode(song, artist, emoji);
    if (!this.tail) {
      this.head = this.tail = node;
    } else {
      node.prev      = this.tail;
      this.tail.next = node;
      this.tail      = node;
    }
    this.size++;
  }

  deleteAt(pos) { // 1-indexed
    if (pos < 1 || pos > this.size) return null;
    let cur = this.head;
    for (let i = 1; i < pos; i++) cur = cur.next;
    if (cur.prev) cur.prev.next = cur.next;
    else          this.head     = cur.next;
    if (cur.next) cur.next.prev = cur.prev;
    else          this.tail     = cur.prev;
    this.size--;
    return cur;
  }

  // Reverse the list in place (shuffle mode)
  reverse() {
    let cur = this.head;
    let tmp = null;
    while (cur) {
      tmp      = cur.prev;
      cur.prev = cur.next;
      cur.next = tmp;
      cur      = cur.prev; // advance (prev is now the old next)
    }
    if (tmp) this.head = tmp.prev; // tmp is last node processed; its prev is the new head
    // recalculate head/tail
    this.tail = this.head;
    let node  = this.head;
    while (node && node.next) { this.tail = node.next; node = node.next; }
  }

  toArray() {
    const arr = [];
    let cur = this.head;
    while (cur) { arr.push(cur); cur = cur.next; }
    return arr;
  }
}

let totalOps    = 0;
let totalShuffs = 0;

const EMOJIS = ['🎵','🎸','🎹','🎷','🎺','🎻','🥁','🎤','🎧','🎼','🪗','🪘','🪕','🎙'];
const SONGS  = [
  ['Blinding Lights','The Weeknd'],['Levitating','Dua Lipa'],
  ['Peaches','Justin Bieber'],['Good 4 U','Olivia Rodrigo'],
  ['Stay','The Kid LAROI'],['Industry Baby','Lil Nas X'],
  ['Shivers','Ed Sheeran'],['Easy On Me','Adele'],
  ['Heat Waves','Glass Animals'],['As It Was','Harry Styles'],
  ['About Damn Time','Lizzo'],['Unholy','Sam Smith']
];
let songIdx = 0;

function pickEmoji() { return EMOJIS[Math.floor(Math.random() * EMOJIS.length)]; }

function getSongInput() {
  let song   = document.getElementById('song-input').value.trim();
  let artist = document.getElementById('artist-input').value.trim();
  if (!song) {
    // auto-fill
    const [s, a] = SONGS[songIdx % SONGS.length]; songIdx++;
    song = s; artist = a;
  }
  if (!artist) artist = 'Unknown Artist';
  document.getElementById('song-input').value  = '';
  document.getElementById('artist-input').value = '';
  return [song, artist, pickEmoji()];
}

function addFront() {
  const [song, artist, emoji] = getSongInput();
  dll.addFront(song, artist, emoji);
  totalOps++;
  log(`addFront: "${song}" by ${artist}`, 'ok');
  toast(`⏮ "${song}" added to front`);
  render();
}

function addEnd() {
  const [song, artist, emoji] = getSongInput();
  dll.addEnd(song, artist, emoji);
  totalOps++;
  log(`addEnd: "${song}" by ${artist}`, 'ok');
  toast(`⏭ "${song}" added to end`);
  render();
}

function deleteByPos() {
  const pos = parseInt(document.getElementById('del-pos').value);
  if (isNaN(pos) || pos < 1) { toast('Enter a valid position', true); return; }
  if (dll.size === 0) { toast('Playlist is empty', true); return; }
  if (pos > dll.size) { toast(`Position ${pos} out of range (size=${dll.size})`, true); return; }
  const removed = dll.deleteAt(pos);
  document.getElementById('del-pos').value = '';
  totalOps++;
  log(`deleteAt(${pos}): removed "${removed.song}"`, 'err');
  toast(`🗑 "${removed.song}" removed`);
  render();
}

function toggleShuffle() {
  if (dll.size < 2) { toast('Add at least 2 songs to shuffle', true); return; }
  dll.reverse();
  shuffled = !shuffled;
  totalOps++; totalShuffs++;
  log(`reverse(): list ${shuffled ? 'reversed' : 'restored'}`, 'shuffle');
  toast(shuffled ? '🔀 Shuffled!' : '▶ Normal order restored');
  render();
}

function resetList() {
  dll.head = dll.tail = null; dll.size = 0;
  shuffled = false; totalOps = 0; totalShuffs = 0; songIdx = 0;
  log('Playlist reset.', 'inf');
  toast('Playlist cleared');
  render();
}

// ── Render ───────────────────────────────────────────────────
function render() {
  renderViz();
  renderPlaylist();
  renderStats();
}

function renderStats() {
  const nodes = dll.toArray();
  document.getElementById('st-size').textContent     = dll.size;
  document.getElementById('st-head').textContent     = dll.head ? trunc(dll.head.song, 8) : 'null';
  document.getElementById('st-tail').textContent     = dll.tail ? trunc(dll.tail.song, 8) : 'null';
  document.getElementById('st-mode').textContent     = shuffled ? 'Shuffle 🔀' : 'Normal ▶';
  document.getElementById('st-ops').textContent      = totalOps;
  document.getElementById('st-shuffles').textContent = totalShuffs;
  document.getElementById('viz-count').textContent   = dll.size + ' node' + (dll.size!==1?'s':'');
  document.getElementById('pl-count').textContent    = dll.size + ' song' + (dll.size!==1?'s':'');
  document.getElementById('shuffle-banner').className = 'shuffle-banner' + (shuffled ? ' active' : '');
  document.getElementById('btn-shuffle').textContent  = shuffled ? '▶ Normal Mode (Un-Reverse)' : '🔀 Shuffle Mode (Reverse)';
}

function renderViz() {
  const nodes  = dll.toArray();
  const empty  = document.getElementById('ll-empty');
  const track  = document.getElementById('ll-track');

  if (!nodes.length) {
    empty.style.display = 'flex';
    track.style.display = 'none';
    return;
  }
  empty.style.display = 'none';
  track.style.display = 'flex';

  let html = `<span class="ll-null">NULL ←</span>`;

  nodes.forEach((node, i) => {
    const isHead = (i === 0);
    const isTail = (i === nodes.length - 1);
    const isBoth = (isHead && isTail);

    let cardCls = 'node-card';
    if (isBoth)       cardCls += ' head-node tail-node';
    else if (isHead)  cardCls += ' head-node';
    else if (isTail)  cardCls += ' tail-node';
    if (shuffled)     cardCls += ' shuffle-hi';

    let tagHtml = '';
    if (isBoth)      tagHtml = `<span class="node-tag tag-both">H+T</span>`;
    else if (isHead) tagHtml = `<span class="node-tag tag-head">HEAD</span>`;
    else if (isTail) tagHtml = `<span class="node-tag tag-tail">TAIL</span>`;

    const prevTxt = isHead ? 'NULL' : '←';
    const nextTxt = isTail ? 'NULL' : '→';

    html += `
      <div class="ll-node">
        <div class="${cardCls}">
          ${tagHtml}
          <div class="node-ptr">
            <span class="ptr-prev">${prevTxt}</span>
            <span class="ptr-next">${nextTxt}</span>
          </div>
          <div class="node-body">
            <div class="node-emoji">${node.emoji}</div>
            <div class="node-title">${esc(node.song)}</div>
            <div class="node-artist">${esc(node.artist)}</div>
            <div class="node-idx">[${i}]</div>
          </div>
        </div>
      </div>`;

    if (i < nodes.length - 1) {
      html += `
        <div class="ll-arrow">
          <div class="arrow-fwd"><div class="arrow-line fwd-line"></div><span>▶</span></div>
          <div class="arrow-bwd"><span>◀</span><div class="arrow-line bwd-line"></div></div>
        </div>`;
    }
  });

  html += `<span class="ll-null">→ NULL</span>`;
  track.innerHTML = html;
}

function renderPlaylist() {
  const body  = document.getElementById('playlist-body');
  const nodes = dll.toArray();
  if (!nodes.length) { body.innerHTML = ''; return; }

  body.innerHTML = nodes.map((node, i) => {
    const isHead = i === 0;
    const isTail = i === nodes.length - 1;
    let cls = 'pl-row';
    if (isHead) cls += ' is-head';
    if (isTail) cls += ' is-tail';
    const tags = [];
    if (isHead) tags.push(`<span class="badge badge-pink">HEAD</span>`);
    if (isTail) tags.push(`<span class="badge badge-purple">TAIL</span>`);
    return `<div class="${cls}">
      <span class="pl-num">${i+1}</span>
      <span class="pl-emoji">${node.emoji}</span>
      <div class="pl-info">
        <div class="pl-title">${esc(node.song)}</div>
        <div class="pl-artist">${esc(node.artist)}</div>
      </div>
      <div class="pl-badge-wrap">${tags.join('')}</div>
    </div>`;
  }).join('');
}

// ── Helpers ──────────────────────────────────────────────────
function trunc(s, n) { return s.length > n ? s.slice(0,n)+'…' : s; }
function esc(s) { return s.replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;'); }

function log(msg, type='') {
  const body = document.getElementById('log-body');
  const div  = document.createElement('div');
  div.className = 'log-entry';
  div.innerHTML = `<span class="log-t">${timeNow()}</span><span class="log-m ${type}">${msg}</span>`;
  body.prepend(div);
  while (body.children.length > 30) body.lastChild.remove();
}

function timeNow() {
  return new Date().toLocaleTimeString('en-IN',{hour:'2-digit',minute:'2-digit',second:'2-digit'});
}

let toastT;
function toast(msg, err=false) {
  const el = document.getElementById('toast');
  el.textContent = msg;
  el.className = 'show' + (err ? ' err' : '');
  clearTimeout(toastT);
  toastT = setTimeout(() => el.className='', 2300);
}
