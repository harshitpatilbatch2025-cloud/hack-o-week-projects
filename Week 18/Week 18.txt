// ── Binary Tree Node ─────────────────────────────────────────
class TreeNode {
  constructor(val) {
    this.val   = val;
    this.left  = null;
    this.right = null;
  }
}

// BST insert (alphabetical order so tree is structured meaningfully)
class BST {
  constructor() { this.root = null; this.size = 0; }

  insert(val) {
    const node = new TreeNode(val);
    if (!this.root) { this.root = node; this.size++; return true; }
    let cur = this.root;
    while (true) {
      const cmp = val.toLowerCase().localeCompare(cur.val.toLowerCase());
      if (cmp === 0) return false; // duplicate
      if (cmp < 0) {
        if (!cur.left)  { cur.left  = node; this.size++; return true; }
        cur = cur.left;
      } else {
        if (!cur.right) { cur.right = node; this.size++; return true; }
        cur = cur.right;
      }
    }
  }

  preOrder(node = this.root, res = []) {
    if (!node) return res;
    res.push(node.val);
    this.preOrder(node.left, res);
    this.preOrder(node.right, res);
    return res;
  }

  inOrder(node = this.root, res = []) {
    if (!node) return res;
    this.inOrder(node.left, res);
    res.push(node.val);
    this.inOrder(node.right, res);
    return res;
  }

  postOrder(node = this.root, res = []) {
    if (!node) return res;
    this.postOrder(node.left, res);
    this.postOrder(node.right, res);
    res.push(node.val);
    return res;
  }
}

const tree = new BST();

// ── Icons by extension/name heuristic ───────────────────────
function iconFor(name) {
  const n = name.toLowerCase();
  if (n.includes('.jpg') || n.includes('.png') || n.includes('.gif') || n.includes('.webp')) return '🖼';
  if (n.includes('.mp3') || n.includes('.wav') || n.includes('music')) return '🎵';
  if (n.includes('.mp4') || n.includes('.mov') || n.includes('video')) return '🎬';
  if (n.includes('.pdf') || n.includes('.doc')) return '📄';
  if (n.includes('.zip') || n.includes('.tar')) return '🗜';
  if (n.includes('.js') || n.includes('.py') || n.includes('.cpp') || n.includes('.c')) return '⚙';
  if (n.includes('.html') || n.includes('.css')) return '🌐';
  if (n.includes('config') || n.includes('.env')) return '🔧';
  if (n.includes('bin') || n.includes('usr') || n.includes('etc') || n.includes('root')) return '🔒';
  return '📁';
}

// ── Insert ───────────────────────────────────────────────────
function insertNode() {
  const inp = document.getElementById('node-input');
  const val = inp.value.trim();
  if (!val) { toast('Enter a name first', true); return; }
  if (!/^[\w\.\-]+$/.test(val)) { toast('Use letters, digits, . or -', true); return; }
  if (!tree.insert(val)) {
    toast(`"${val}" already exists`, true);
    log(`Duplicate rejected: ${val}`, 'err');
    return;
  }
  inp.value = '';
  log(`Inserted: ${val} (size=${tree.size})`, 'ok');
  toast(`${iconFor(val)} ${val} inserted`);
  renderTree();
  updateCount();
}

// ── Traversals ───────────────────────────────────────────────
let lastTraversal = null;

function runTraversal(type) {
  if (!tree.root) { toast('Tree is empty', true); return; }
  let result;
  let label, cls;
  if (type === 'pre')  { result = tree.preOrder();  label = 'Pre-Order  (Root → Left → Right)'; cls = 'pre';  }
  if (type === 'in')   { result = tree.inOrder();   label = 'In-Order   (Left → Root → Right)'; cls = 'in';  }
  if (type === 'post') { result = tree.postOrder(); label = 'Post-Order (Left → Right → Root)'; cls = 'post'; }

  lastTraversal = { type, result };
  log(`${label}: ${result.join(' → ')}`, 'inf');
  renderTraversal(result, label, cls);
  highlightTraversal(result, cls);
}

function renderTraversal(result, label, cls) {
  const body = document.getElementById('trav-body');
  // keep previous results (up to 3)
  const existing = body.querySelectorAll('.trav-row');
  if (existing.length >= 3) existing[0].remove();

  const div = document.createElement('div');
  div.className = 'trav-row';
  div.innerHTML = `<div class="trav-label ${cls}">${label}</div>
    <div class="trav-nodes">${result.map(n => `<span class="tn ${cls}">${iconFor(n)} ${n}</span>`).join('')}</div>`;
  // remove empty placeholder
  body.querySelector('.trav-empty')?.remove();
  body.appendChild(div);
}

// ── Tree SVG Rendering ───────────────────────────────────────
const NODE_R = 26;
const H_GAP  = 14;
const V_GAP  = 68;

let highlightedNodes = new Set();
let highlightClass   = '';

function highlightTraversal(result, cls) {
  highlightedNodes = new Set(result);
  highlightClass   = cls;
  renderTree();
  // animate sequence
  result.forEach((val, i) => {
    setTimeout(() => {
      highlightedNodes = new Set(result.slice(0, i+1));
      renderTree();
    }, i * 220);
  });
}

function computeLayout(node, depth=0, counter={v:0}) {
  if (!node) return;
  computeLayout(node.left, depth+1, counter);
  node._x = counter.v * (NODE_R * 2 + H_GAP);
  node._y = depth * V_GAP;
  counter.v++;
  computeLayout(node.right, depth+1, counter);
}

function renderTree() {
  const wrap = document.getElementById('tree-canvas-wrap');
  const svg  = document.getElementById('tree-svg');

  if (!tree.root) {
    svg.innerHTML = `<text x="50%" y="50%" text-anchor="middle" dominant-baseline="middle"
      font-family="Fira Code" font-size="13" fill="#1e2d3d" letter-spacing="2">EMPTY TREE</text>`;
    svg.setAttribute('width', '100%');
    svg.setAttribute('height', '200');
    return;
  }

  computeLayout(tree.root);

  // collect all nodes
  const nodes = [];
  const edges = [];
  function collect(node, parent=null) {
    if (!node) return;
    nodes.push(node);
    if (parent) edges.push({ x1: parent._x, y1: parent._y, x2: node._x, y2: node._y });
    collect(node.left, node);
    collect(node.right, node);
  }
  collect(tree.root);

  // bounds
  const minX = Math.min(...nodes.map(n=>n._x));
  const maxX = Math.max(...nodes.map(n=>n._x));
  const maxY = Math.max(...nodes.map(n=>n._y));
  const pad  = NODE_R + 20;
  const W    = Math.max(600, maxX - minX + pad*2);
  const H    = maxY + pad*2 + 20;
  const offX = -minX + pad;
  const offY = pad;

  let out = '';

  // edges
  edges.forEach(e => {
    out += `<line x1="${e.x1+offX}" y1="${e.y1+offY}" x2="${e.x2+offX}" y2="${e.y2+offY}"
      stroke="#1e2d3d" stroke-width="1.5" stroke-dasharray="4 3"/>`;
  });

  // nodes
  nodes.forEach(node => {
    const cx = node._x + offX;
    const cy = node._y + offY;
    const isRoot = (node === tree.root);
    const isHL   = highlightedNodes.has(node.val);

    let stroke = '#1e2d3d', strokeW = 1.5, fill = '#0d1117', glowId = '';
    let textCol = '#c9d8e8';

    if (isHL) {
      if (highlightClass === 'pre')  { stroke = '#3ddc84'; fill = 'rgba(61,220,132,0.12)'; textCol='#3ddc84'; }
      if (highlightClass === 'in')   { stroke = '#f0c040'; fill = 'rgba(240,192,64,0.12)'; textCol='#f0c040'; }
      if (highlightClass === 'post') { stroke = '#b388ff'; fill = 'rgba(179,136,255,0.12)'; textCol='#b388ff'; }
      strokeW = 2.5;
    } else if (isRoot) {
      stroke = '#00d4ff'; fill = 'rgba(0,212,255,0.1)'; strokeW = 2; textCol='#00d4ff';
    }

    // node circle
    out += `<circle cx="${cx}" cy="${cy}" r="${NODE_R}" fill="${fill}" stroke="${stroke}" stroke-width="${strokeW}"/>`;

    // icon
    const ico = iconFor(node.val);
    out += `<text x="${cx}" y="${cy-7}" text-anchor="middle" dominant-baseline="middle"
      font-size="12">${ico}</text>`;

    // name (truncated)
    const label = node.val.length > 9 ? node.val.slice(0,8)+'…' : node.val;
    out += `<text x="${cx}" y="${cy+10}" text-anchor="middle" dominant-baseline="middle"
      font-family="Fira Code" font-size="8.5" font-weight="500" fill="${textCol}">${label}</text>`;

    // root badge
    if (isRoot) {
      out += `<text x="${cx}" y="${cy - NODE_R - 7}" text-anchor="middle"
        font-family="Fira Code" font-size="7" fill="#00d4ff" letter-spacing="2">ROOT</text>`;
    }
  });

  svg.innerHTML = out;
  svg.setAttribute('width',  W);
  svg.setAttribute('height', H);
  wrap.style.minHeight = Math.max(320, H+20)+'px';
}

function updateCount() {
  document.getElementById('node-count').textContent = tree.size + (tree.size===1?' node':' nodes');
}

function resetTree() {
  tree.root = null; tree.size = 0;
  highlightedNodes = new Set(); highlightClass = '';
  document.getElementById('trav-body').innerHTML = '<div class="trav-empty">Run a traversal to see results here.</div>';
  renderTree(); updateCount();
  log('Tree reset.', 'err');
  toast('Tree cleared');
}

// ── Log ──────────────────────────────────────────────────────
function log(msg, type='') {
  const body = document.getElementById('log-body');
  const div  = document.createElement('div');
  div.className = 'log-entry';
  div.innerHTML = `<span class="log-t">${timeNow()}</span><span class="log-m ${type}">${msg}</span>`;
  body.prepend(div);
  while (body.children.length > 25) body.lastChild.remove();
}

function timeNow() {
  return new Date().toLocaleTimeString('en-IN',{hour:'2-digit',minute:'2-digit',second:'2-digit'});
}

let toastT;
function toast(msg, err=false) {
  const el = document.getElementById('toast');
  el.textContent = msg;
  el.className = 'show' + (err ? ' err' : '');
  clearTimeout(toastT);
  toastT = setTimeout(() => el.className='', 2200);
}

  if (e.key === 'Enter') insertNode();
});
