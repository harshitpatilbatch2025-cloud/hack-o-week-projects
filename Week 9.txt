<script>
  // Date line
  const d = new Date();
  document.getElementById('dateLine').textContent =
    d.toLocaleDateString('en-IN', { weekday:'long', year:'numeric', month:'long', day:'numeric' });

  // Temperature data (°F)
  const data = {
    delhi:  { cur: 81.9, hi: 92.8, forecast: [92.8, 92.3, 91.7, 92.4, 95.7] },
    mumbai: { cur: 81.2, hi: 86.5, forecast: [86.5, 88.7, 89.3, 88.4, 88.5] },
    blr:    { cur: 80.2, hi: 91.3, forecast: [91.3, 90.6, 92.3, 91.8, 91.0] },
  };

  let unit = 'F';

  function fToC(f) { return ((f - 32) * 5 / 9).toFixed(1); }
  function fmt(f)  { return unit === 'F' ? Math.round(f) : Math.round(fToC(f)); }
  function sym()   { return unit === 'F' ? '°F' : '°C'; }

  function updateCards() {
    // Delhi
    document.getElementById('delhi-temp').textContent = fmt(data.delhi.cur);
    document.getElementById('delhi-unit').textContent = sym();
    document.getElementById('delhi-hi').textContent   = fmt(data.delhi.hi) + '°';
    ['d-f1','d-f2','d-f3','d-f4','d-f5'].forEach((id, i) => {
      document.getElementById(id).textContent = fmt(data.delhi.forecast[i]) + '°';
    });
    // Mumbai
    document.getElementById('mumbai-temp').textContent = fmt(data.mumbai.cur);
    document.getElementById('mumbai-unit').textContent = sym();
    document.getElementById('mumbai-hi').textContent   = fmt(data.mumbai.hi) + '°';
    ['m-f1','m-f2','m-f3','m-f4','m-f5'].forEach((id, i) => {
      document.getElementById(id).textContent = fmt(data.mumbai.forecast[i]) + '°';
    });
    // Bengaluru
    document.getElementById('blr-temp').textContent = fmt(data.blr.cur);
    document.getElementById('blr-unit').textContent = sym();
    document.getElementById('blr-hi').textContent   = fmt(data.blr.hi) + '°';
    ['b-f1','b-f2','b-f3','b-f4','b-f5'].forEach((id, i) => {
      document.getElementById(id).textContent = fmt(data.blr.forecast[i]) + '°';
    });
  }

  function setUnit(u) {
    unit = u;
    document.getElementById('btnF').classList.toggle('active', u === 'F');
    document.getElementById('btnC').classList.toggle('active', u === 'C');
    updateCards();
  }

  updateCards();
</script>
