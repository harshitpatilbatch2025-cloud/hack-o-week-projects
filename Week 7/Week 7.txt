<script>
  // Your 7 Choice Tasks
  const defaultTasks = [
    { id: 1, text: "Deep Work: 2-hour focus session", completed: false },
    { id: 2, text: "Review and respond to urgent emails", completed: true },
    { id: 3, text: "Morning 30-minute workout", completed: false },
    { id: 4, text: "Update project timeline and milestones", completed: false },
    { id: 5, text: "Research new industry trends", completed: false },
    { id: 6, text: "Quick 15-minute desk organization", completed: true },
    { id: 7, text: "Prepare agenda for tomorrow's meeting", completed: false }
  ];

  let tasks = JSON.parse(localStorage.getItem('simple_focus_tasks')) || defaultTasks;

  function init() {
    const today = new Date();
    document.getElementById('dateDisplay').innerText = today.toLocaleDateString('en-US', { weekday: 'long', month: 'short', day: 'numeric' });
    render();
  }

  function handleEnter(e) {
    if (e.key === 'Enter' && e.target.value.trim() !== '') {
      addTask(e.target.value);
      e.target.value = '';
    }
  }

  function addTask(text) {
    tasks.unshift({ id: Date.now(), text, completed: false });
    saveAndRender();
  }

  function toggleTask(id) {
    tasks = tasks.map(t => t.id === id ? {...t, completed: !t.completed} : t);
    saveAndRender();
  }

  function deleteTask(e, id) {
    e.stopPropagation();
    tasks = tasks.filter(t => t.id !== id);
    saveAndRender();
  }

  function saveAndRender() {
    localStorage.setItem('simple_focus_tasks', JSON.stringify(tasks));
    render();
  }

  function render() {
    const list = document.getElementById('taskList');
    list.innerHTML = '';
    let completed = 0;

    if (tasks.length === 0) {
      list.innerHTML = '<div class="empty-msg">All clear! Add a task to start.</div>';
    }

    tasks.forEach(task => {
      if(task.completed) completed++;
      const item = document.createElement('div');
      item.className = `task-item ${task.completed ? 'completed' : ''}`;
      item.onclick = () => toggleTask(task.id);
      
      item.innerHTML = `
        <div class="checkbox">${task.completed ? '✓' : ''}</div>
        <span class="task-text">${task.text}</span>
        <button class="del-btn" onclick="deleteTask(event, ${task.id})">×</button>
      `;
      list.appendChild(item);
    });

    const percent = tasks.length === 0 ? 0 : Math.round((completed / tasks.length) * 100);
    document.getElementById('progressHeader').innerText = percent + '%';
    document.getElementById('progressBar').style.width = percent + '%';
  }

  init();
</script>
