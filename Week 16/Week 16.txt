<script>
// ── Data structures ──────────────────────────────────────────
let tasks = [];          // Array of {id, text, done}
let undoStack = [];      // Stack of action objects
let redoStack = [];      // Stack of undone actions
let nextId = 1;

// Action types: 'ADD', 'DELETE', 'EDIT', 'TOGGLE'
// Each action stores enough to undo and redo it.

// ── Core task operations (do NOT call these directly for user actions) ──
function _addTask(id, text) {
  tasks.push({ id, text, done: false });
  renderTasks();
}
function _removeTask(id) {
  tasks = tasks.filter(t => t.id !== id);
  renderTasks();
}
function _editTask(id, newText) {
  const t = tasks.find(t => t.id === id);
  if (t) { t.text = newText; renderTasks(); }
}
function _toggleTask(id, done) {
  const t = tasks.find(t => t.id === id);
  if (t) { t.done = done; renderTasks(); }
}

// ── User-facing actions (push onto undo stack) ──
function addTask() {
  const input = document.getElementById('task-input');
  const text = input.value.trim();
  if (!text) { toast('⚠ Please enter a task'); return; }
  const id = nextId++;
  _addTask(id, text);
  input.value = '';
  pushUndo({ type: 'ADD', id, text });
  flashItem(id, 'add');
  toast('Task added');
  updateControls();
}

function deleteTask(id) {
  const t = tasks.find(t => t.id === id);
  if (!t) return;
  const saved = { ...t };
  _removeTask(id);
  pushUndo({ type: 'DELETE', id, text: saved.text, done: saved.done });
  toast('Task deleted');
  updateControls();
}

function editTask(id, newText, oldText) {
  if (!newText || newText === oldText) return;
  _editTask(id, newText);
  pushUndo({ type: 'EDIT', id, oldText, newText });
  flashItem(id, 'edit');
  toast('Task edited');
  updateControls();
}

function toggleTask(id) {
  const t = tasks.find(t => t.id === id);
  if (!t) return;
  const oldDone = t.done;
  _toggleTask(id, !oldDone);
  pushUndo({ type: 'TOGGLE', id, oldDone, newDone: !oldDone });
  updateControls();
}

function clearAll() {
  if (!tasks.length) return;
  const snapshot = [...tasks.map(t => ({ ...t }))];
  tasks = [];
  renderTasks();
  pushUndo({ type: 'CLEAR', snapshot });
  toast('All tasks cleared');
  updateControls();
}

// ── Stack operations ──
function pushUndo(action) {
  undoStack.push(action);   // push onto undo stack
  redoStack = [];            // any new action wipes redo stack
  renderStacks();
}

function undo() {
  if (!undoStack.length) return;
  const action = undoStack.pop();   // pop from undo stack
  applyUndo(action);
  redoStack.push(action);            // push onto redo stack
  toast('↩ Undone: ' + labelFor(action));
  renderStacks();
  updateControls();
}

function redo() {
  if (!redoStack.length) return;
  const action = redoStack.pop();   // pop from redo stack
  applyRedo(action);
  undoStack.push(action);            // push back onto undo stack
  toast('↪ Redone: ' + labelFor(action));
  renderStacks();
  updateControls();
}

function applyUndo(a) {
  if (a.type === 'ADD')    { _removeTask(a.id); }
  if (a.type === 'DELETE') { tasks.push({ id: a.id, text: a.text, done: a.done }); renderTasks(); }
  if (a.type === 'EDIT')   { _editTask(a.id, a.oldText); }
  if (a.type === 'TOGGLE') { _toggleTask(a.id, a.oldDone); }
  if (a.type === 'CLEAR')  { tasks = a.snapshot.map(t => ({ ...t })); renderTasks(); }
}

function applyRedo(a) {
  if (a.type === 'ADD')    { _addTask(a.id, a.text); flashItem(a.id, 'add'); }
  if (a.type === 'DELETE') { _removeTask(a.id); }
  if (a.type === 'EDIT')   { _editTask(a.id, a.newText); flashItem(a.id, 'edit'); }
  if (a.type === 'TOGGLE') { _toggleTask(a.id, a.newDone); }
  if (a.type === 'CLEAR')  { tasks = []; renderTasks(); }
}

function labelFor(a) {
  if (a.type === 'ADD')    return `Add "${truncate(a.text)}"`;
  if (a.type === 'DELETE') return `Delete "${truncate(a.text)}"`;
  if (a.type === 'EDIT')   return `Edit "${truncate(a.oldText)}"`;
  if (a.type === 'TOGGLE') return `Toggle task`;
  if (a.type === 'CLEAR')  return `Clear all`;
  return a.type;
}
function truncate(s, n=20) { return s.length > n ? s.slice(0,n)+'…' : s; }

// ── Rendering ──
function renderTasks() {
  const list = document.getElementById('task-list');
  document.getElementById('task-count').textContent =
    tasks.length + (tasks.length === 1 ? ' item' : ' items');

  if (!tasks.length) {
    list.innerHTML = '<div class="empty-state">No tasks. Add one or undo ↩</div>';
    return;
  }

  list.innerHTML = tasks.map(t => `
    <div class="task-item" id="item-${t.id}">
      <div class="task-check ${t.done ? 'done' : ''}" onclick="toggleTask(${t.id})"></div>
      <span class="task-text ${t.done ? 'done' : ''}" id="text-${t.id}">${escHtml(t.text)}</span>
      <button class="task-edit-btn" onclick="startEdit(${t.id})" title="Edit">✎</button>
      <button class="task-del-btn" onclick="deleteTask(${t.id})" title="Delete">✕</button>
    </div>
  `).join('');
}

function startEdit(id) {
  const span = document.getElementById('text-' + id);
  const t = tasks.find(t => t.id === id);
  if (!t) return;
  const oldText = t.text;
  span.innerHTML = `<input class="task-edit-input" id="edit-input-${id}" value="${escHtml(oldText)}" />`;
  const inp = document.getElementById('edit-input-' + id);
  inp.focus(); inp.select();
  inp.addEventListener('blur', () => {
    const newText = inp.value.trim();
    editTask(id, newText || oldText, oldText);
  });
  inp.addEventListener('keydown', e => {
    if (e.key === 'Enter') inp.blur();
    if (e.key === 'Escape') { inp.value = oldText; inp.blur(); }
  });
}

function renderStacks() {
  renderOneStack('undo', undoStack, 'undo-frame');
  renderOneStack('redo', redoStack, 'redo-frame');
  document.getElementById('undo-size').textContent = undoStack.length;
  document.getElementById('redo-size').textContent = redoStack.length;
}

function renderOneStack(name, stack, cls) {
  const viz = document.getElementById(name + '-stack-viz');
  const ptr = document.getElementById(name + '-pointer');
  if (!stack.length) {
    viz.innerHTML = '<div class="stack-empty">— empty —</div>';
    ptr.textContent = '';
    return;
  }
  // Show last 6 frames max (top = last element)
  const visible = stack.slice(-6);
  viz.innerHTML = visible.map((a, i) => {
    const isTop = (i === visible.length - 1);
    const tagCls = {ADD:'tag-add', DELETE:'tag-del', EDIT:'tag-edit', TOGGLE:'tag-done', CLEAR:'tag-del'}[a.type] || 'tag-add';
    const label = labelFor(a);
    return `<div class="stack-frame ${cls} ${isTop ? 'top-frame' : ''}">
      <span class="tag ${tagCls}">${a.type}</span>
      <span style="overflow:hidden;text-overflow:ellipsis;white-space:nowrap">${escHtml(label)}</span>
    </div>`;
  }).join('');
  ptr.textContent = stack.length > 6 ? `↑ +${stack.length - 6} more · top of stack` : '↑ top of stack';
}

function updateControls() {
  document.getElementById('undo-btn').disabled = !undoStack.length;
  document.getElementById('redo-btn').disabled = !redoStack.length;
}

function flashItem(id, type) {
  setTimeout(() => {
    const el = document.getElementById('item-' + id);
    if (el) {
      el.classList.remove('flash-add', 'flash-del');
      void el.offsetWidth;
      el.classList.add(type === 'add' ? 'flash-add' : type === 'del' ? 'flash-del' : 'flash-add');
    }
  }, 10);
}

let toastTimer;
function toast(msg) {
  const el = document.getElementById('toast');
  el.textContent = msg;
  el.classList.add('show');
  clearTimeout(toastTimer);
  toastTimer = setTimeout(() => el.classList.remove('show'), 2000);
}

function escHtml(s) {
  return s.replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;').replace(/"/g,'&quot;');
}

// Keyboard shortcuts
document.addEventListener('keydown', e => {
  if ((e.ctrlKey || e.metaKey) && e.key === 'z' && !e.shiftKey) { e.preventDefault(); undo(); }
  if ((e.ctrlKey || e.metaKey) && (e.key === 'y' || (e.key === 'z' && e.shiftKey))) { e.preventDefault(); redo(); }
});
document.getElementById('task-input').addEventListener('keydown', e => {
  if (e.key === 'Enter') addTask();
});

// Seed with sample tasks
(function seed() {
  ['Buy groceries', 'Read chapter 3', 'Fix login bug'].forEach(t => {
    const id = nextId++;
    _addTask(id, t);
    undoStack.push({ type: 'ADD', id, text: t });
  });
  renderStacks();
  updateControls();
})();
</script>
