<script>
// ── Circular Queue ───────────────────────────────────────────
const CAPACITY = 6;
let queue = new Array(CAPACITY).fill(null);
let front = 0;
let rear = -1;
let size = 0;
let totalServed = 0;

const NAMES = [
  'Arjun Mehta','Priya Sharma','Rahul Gupta','Sneha Patel',
  'Vikram Rao','Anjali Nair','Suresh Kumar','Deepa Iyer',
  'Kiran Joshi','Meera Pillai','Arun Verma','Pooja Singh',
  'Rohan Das','Lakshmi Menon','Nikhil Bhat','Sunita Yadav'
];
let nameIdx = 0;
function nextName() {
  const n = NAMES[nameIdx % NAMES.length];
  nameIdx++;
  return n;
}

function isEmpty() { return size === 0; }
function isFull()  { return size === CAPACITY; }

function enqueue() {
  if (isFull()) {
    toast('Queue is FULL — please wait for service', true);
    log('Queue full. Customer turned away.', 'err');
    return;
  }
  rear = (rear + 1) % CAPACITY;
  const name = nextName();
  const ts   = timeNow();
  queue[rear] = { name, ts };
  size++;
  log(`Enqueued: ${name} → slot[${rear}]`, 'enq');
  toast(`${name} joined the queue`);
  render();
}

function dequeue() {
  if (isEmpty()) {
    toast('Queue is EMPTY', true);
    log('Dequeue attempted on empty queue.', 'err');
    return;
  }
  const served = queue[front];
  queue[front] = null;
  front = (front + 1) % CAPACITY;
  size--;
  totalServed++;
  log(`Dequeued: ${served.name} (served) ← slot[${(front - 1 + CAPACITY) % CAPACITY}]`, 'deq');
  toast(`${served.name} served ✓`);
  render();
}

function resetQueue() {
  queue = new Array(CAPACITY).fill(null);
  front = 0; rear = -1; size = 0; totalServed = 0; nameIdx = 0;
  log('Queue reset.', '');
  toast('Queue reset');
  render();
}

// ── Render ───────────────────────────────────────────────────
function render() {
  renderSVG();
  renderList();
  renderStats();
}

function renderStats() {
  document.getElementById('stat-cap').textContent   = CAPACITY;
  document.getElementById('stat-size').textContent  = size;
  document.getElementById('stat-front').textContent = isEmpty() ? '—' : front;
  document.getElementById('stat-rear').textContent  = isEmpty() ? '—' : rear;
  document.getElementById('stat-served').textContent = totalServed;

  document.getElementById('i-cap').textContent   = CAPACITY;
  document.getElementById('i-front').textContent = isEmpty() ? '—' : front;
  document.getElementById('i-rear').textContent  = isEmpty() ? '—' : rear;
  document.getElementById('i-size').textContent  = size;
  document.getElementById('i-empty').textContent = isEmpty() ? 'true' : 'false';
  document.getElementById('i-full').textContent  = isFull()  ? 'true' : 'false';
  document.getElementById('i-next').textContent  = isEmpty() ? '—' : `(${rear}+1) % ${CAPACITY} = ${(rear+1)%CAPACITY}`;

  const pill = document.getElementById('status-pill');
  if (isFull()) { pill.textContent = 'FULL'; pill.className = 'pill red'; }
  else { pill.textContent = 'OPEN'; pill.className = 'pill green'; }

  document.getElementById('stat-size').className = 'stat-value ' + (isFull() ? 'red' : 'green');

  document.getElementById('btn-deq').disabled = isEmpty();
  document.getElementById('btn-enq').disabled = isFull();
}

function renderList() {
  const el = document.getElementById('queue-list');
  if (isEmpty()) { el.innerHTML = '<div class="empty-msg">— No customers in queue —</div>'; return; }

  let html = '';
  for (let i = 0; i < size; i++) {
    const idx  = (front + i) % CAPACITY;
    const c    = queue[idx];
    const isF  = (idx === front);
    const isR  = (idx === rear);
    const isFR = (isF && isR);
    const isServing = (i === 0);

    let slotCls = 'slot-mid', slotLabel = idx;
    if (isFR) { slotCls = 'slot-both'; slotLabel = idx; }
    else if (isF) { slotCls = 'slot-front'; }
    else if (isR) { slotCls = 'slot-rear'; }

    const tag = isServing
      ? '<span class="row-tag tag-serving">Serving</span>'
      : '<span class="row-tag tag-waiting">Waiting</span>';

    html += `<div class="queue-row ${isServing ? 'serving' : 'waiting'}">
      <div class="row-slot ${slotCls}" title="${isFR?'front+rear':isF?'front':isR?'rear':''}">${slotLabel}</div>
      <span class="row-name ${isServing?'serving':''}">${c.name}</span>
      <span class="row-time">${c.ts}</span>
      ${tag}
    </div>`;
  }
  el.innerHTML = html;
}

// ── SVG Circular Ring ────────────────────────────────────────
function renderSVG() {
  const svg = document.getElementById('queue-svg');
  const R = 118, r = 34, cx = 0, cy = 0;
  const gold = '#d4a437', blue = '#5ab0d8', green = '#4ec94e', muted = '#4a3518', dim = '#2d2210';

  let out = '';
  // Outer decorative ring
  out += `<circle cx="0" cy="0" r="${R+14}" fill="none" stroke="${muted}" stroke-width="1" stroke-dasharray="4 6"/>`;

  for (let i = 0; i < CAPACITY; i++) {
    const angle = (i / CAPACITY) * 2 * Math.PI - Math.PI / 2;
    const x = Math.round(R * Math.cos(angle));
    const y = Math.round(R * Math.sin(angle));

    const isF = (!isEmpty() && i === front);
    const isR = (!isEmpty() && i === rear);
    const occupied = queue[i] !== null;

    let fill   = occupied ? 'rgba(212,164,55,0.12)' : dim;
    let stroke = occupied ? gold : muted;
    let sw     = 1.5;
    if (isF && isR) { stroke = green; sw = 2.5; fill = 'rgba(78,201,78,0.15)'; }
    else if (isF)   { stroke = gold;  sw = 2.5; fill = 'rgba(212,164,55,0.18)'; }
    else if (isR)   { stroke = blue;  sw = 2.5; fill = 'rgba(90,176,216,0.15)'; }

    // Draw spoke
    out += `<line x1="0" y1="0" x2="${x}" y2="${y}" stroke="${muted}" stroke-width="0.5" stroke-dasharray="3 5"/>`;

    // Slot circle
    out += `<circle cx="${x}" cy="${y}" r="${r}" fill="${fill}" stroke="${stroke}" stroke-width="${sw}"/>`;

    // Slot index
    out += `<text x="${x}" y="${y - 10}" text-anchor="middle" dominant-baseline="middle"
      font-family="IBM Plex Mono" font-size="9" fill="${muted}" letter-spacing="1">slot[${i}]</text>`;

    // Customer name (short)
    if (queue[i]) {
      const short = queue[i].name.split(' ')[0];
      out += `<text x="${x}" y="${y + 2}" text-anchor="middle" dominant-baseline="middle"
        font-family="IBM Plex Mono" font-size="9.5" font-weight="600"
        fill="${isF || isR ? '#f0c060' : '#e8d5a8'}">${short}</text>`;
    } else {
      out += `<text x="${x}" y="${y + 2}" text-anchor="middle" dominant-baseline="middle"
        font-family="IBM Plex Mono" font-size="9" fill="${muted}">—</text>`;
    }

    // Pointer labels
    if (isF && isR) {
      out += `<text x="${x}" y="${y + 18}" text-anchor="middle"
        font-family="IBM Plex Mono" font-size="7.5" fill="${green}" letter-spacing="1">F+R</text>`;
    } else if (isF) {
      out += `<text x="${x}" y="${y + 18}" text-anchor="middle"
        font-family="IBM Plex Mono" font-size="7.5" fill="${gold}" letter-spacing="1">FRONT</text>`;
    } else if (isR) {
      out += `<text x="${x}" y="${y + 18}" text-anchor="middle"
        font-family="IBM Plex Mono" font-size="7.5" fill="${blue}" letter-spacing="1">REAR</text>`;
    }
  }

  // Center info
  out += `<circle cx="0" cy="0" r="52" fill="${dim}" stroke="${muted}" stroke-width="1"/>`;
  out += `<text x="0" y="-18" text-anchor="middle" font-family="Playfair Display" font-size="13" font-weight="700" fill="${gold}">${size}/${CAPACITY}</text>`;
  out += `<text x="0" y="0" text-anchor="middle" font-family="IBM Plex Mono" font-size="8" fill="${muted}" letter-spacing="2">IN QUEUE</text>`;
  const statusTxt = isEmpty() ? 'EMPTY' : isFull() ? 'FULL' : 'ACTIVE';
  const statusCol = isEmpty() ? muted : isFull() ? '#e05040' : green;
  out += `<text x="0" y="16" text-anchor="middle" font-family="IBM Plex Mono" font-size="8" font-weight="600" fill="${statusCol}" letter-spacing="2">${statusTxt}</text>`;

  svg.innerHTML = out;
}

// ── Log ──────────────────────────────────────────────────────
function log(msg, type='') {
  const wrap = document.getElementById('log-wrap');
  const div  = document.createElement('div');
  div.className = 'log-entry';
  div.innerHTML = `<span class="log-time">${timeNow()}</span><span class="log-msg ${type}">${msg}</span>`;
  wrap.prepend(div);
  if (wrap.children.length > 30) wrap.lastChild.remove();
}

function timeNow() {
  return new Date().toLocaleTimeString('en-IN', {hour:'2-digit',minute:'2-digit',second:'2-digit'});
}

// ── Toast ────────────────────────────────────────────────────
let toastT;
function toast(msg, err=false) {
  const el = document.getElementById('toast');
  el.textContent = msg;
  el.className   = 'show' + (err ? ' error' : '');
  clearTimeout(toastT);
  toastT = setTimeout(() => el.className = '', 2200);
}
