<script>
// ── Data ─────────────────────────────────────────────────────
let scores  = [];       // Array of {name, score}
let freq    = new Array(101).fill(0);  // freq[0..100]
let opCount = 0;

// ── Core operations ──────────────────────────────────────────

// Add a score: push to array, increment freq[score]
function _add(name, score) {
  scores.push({ name, score });
  freq[score]++;
  opCount++;
}

// Remove by index: splice from array, decrement freq[score]
function _remove(idx) {
  const removed = scores[idx];
  freq[removed.score]--;
  scores.splice(idx, 1);
  opCount++;
  return removed;
}

// Linear traversal: find min
function findMin() {
  if (!scores.length) return null;
  let minVal = scores[0].score;
  for (let i = 1; i < scores.length; i++) {
    if (scores[i].score < minVal) minVal = scores[i].score;
  }
  return minVal;
}

// Linear traversal: find max
function findMax() {
  if (!scores.length) return null;
  let maxVal = scores[0].score;
  for (let i = 1; i < scores.length; i++) {
    if (scores[i].score > maxVal) maxVal = scores[i].score;
  }
  return maxVal;
}

// Linear traversal over freq[]: find mode (most frequent score)
function findMode() {
  if (!scores.length) return null;
  let modeVal = -1, modeFreq = 0;
  for (let s = 0; s <= 100; s++) {
    if (freq[s] > modeFreq) { modeFreq = freq[s]; modeVal = s; }
  }
  return { val: modeVal, freq: modeFreq };
}

// ── User actions ─────────────────────────────────────────────
function addScore() {
  const nameEl  = document.getElementById('inp-name');
  const scoreEl = document.getElementById('inp-score');
  const name    = nameEl.value.trim() || autoName();
  const score   = parseInt(scoreEl.value);

  if (isNaN(score) || score < 0 || score > 100) {
    toast('Score must be 0–100', true);
    return;
  }
  _add(name, score);
  nameEl.value  = '';
  scoreEl.value = '';
  log(`scores.push({name:"${name}", score:${score}}) | freq[${score}]++ → ${freq[score]}`, 'ok');
  toast(`✓ ${name}: ${score}`);
  render();
}

function batchAdd() {
  const inp = document.getElementById('inp-batch');
  const raw = inp.value.split(',').map(s => s.trim()).filter(Boolean);
  if (!raw.length) { toast('Enter comma-separated scores', true); return; }
  let added = 0;
  raw.forEach(s => {
    const v = parseInt(s);
    if (!isNaN(v) && v >= 0 && v <= 100) {
      _add(autoName(), v);
      added++;
    }
  });
  inp.value = '';
  log(`Batch insert: ${added} scores added. Array length now ${scores.length}.`, 'ok');
  toast(`✓ ${added} scores added`);
  render();
}

function removeScore(idx) {
  const r = _remove(idx);
  log(`scores.splice(${idx},1) | removed "${r.name}" score=${r.score} | freq[${r.score}]-- → ${freq[r.score]}`, 'err');
  toast(`Removed ${r.name}: ${r.score}`);
  render();
}

function clearAll() {
  scores = []; freq = new Array(101).fill(0); opCount = 0;
  log('scores[] cleared. freq[0..100] reset to 0.', 'inf');
  toast('All data cleared');
  render();
}

// ── Render ───────────────────────────────────────────────────
function render() {
  const minVal  = findMin();
  const maxVal  = findMax();
  const modeRes = findMode();

  renderStats(minVal, maxVal, modeRes);
  renderArray(minVal, maxVal, modeRes);
  renderFreq(modeRes);
  renderList(minVal, maxVal, modeRes);
  renderCounts();
}

function renderStats(minVal, maxVal, modeRes) {
  if (scores.length === 0) {
    document.getElementById('stat-min').textContent      = '—';
    document.getElementById('stat-min-name').textContent = 'No data';
    document.getElementById('stat-max').textContent      = '—';
    document.getElementById('stat-max-name').textContent = 'No data';
    document.getElementById('stat-mode').textContent     = '—';
    document.getElementById('stat-mode-freq').textContent= 'No data';
    return;
  }
  // min
  const minStudents = scores.filter(s=>s.score===minVal).map(s=>s.name);
  document.getElementById('stat-min').textContent      = minVal;
  document.getElementById('stat-min-name').textContent = minStudents.slice(0,2).join(', ') + (minStudents.length>2?'…':'');

  // max
  const maxStudents = scores.filter(s=>s.score===maxVal).map(s=>s.name);
  document.getElementById('stat-max').textContent      = maxVal;
  document.getElementById('stat-max-name').textContent = maxStudents.slice(0,2).join(', ') + (maxStudents.length>2?'…':'');

  // mode
  document.getElementById('stat-mode').textContent      = modeRes.val;
  document.getElementById('stat-mode-freq').textContent = `appeared ${modeRes.freq}×`;
}

function renderArray(minVal, maxVal, modeRes) {
  const track = document.getElementById('array-track');
  document.getElementById('arr-chip').textContent = `length: ${scores.length}`;

  if (!scores.length) {
    track.innerHTML = '<div class="array-empty">scores[] is empty</div>';
    return;
  }

  track.innerHTML = scores.map((s, i) => {
    const isMin  = s.score === minVal;
    const isMax  = s.score === maxVal;
    const isMode = modeRes && s.score === modeRes.val && modeRes.freq > 1;

    let cls = 'cell-box normal';
    if (isMin && isMode) cls = 'cell-box is-min is-mode';
    else if (isMax && isMode) cls = 'cell-box is-max is-mode';
    else if (isMin)  cls = 'cell-box is-min';
    else if (isMax)  cls = 'cell-box is-max';
    else if (isMode) cls = 'cell-box is-mode';

    const flags = [];
    if (isMin)  flags.push(`<span class="cell-flag flag-min">MIN</span>`);
    if (isMax)  flags.push(`<span class="cell-flag flag-max" style="top:-22px">MAX</span>`);
    if (isMode && modeRes.freq > 1) flags.push(`<span class="cell-flag flag-mode" style="top:${isMin||isMax?'-34px':'-10px'}">MODE</span>`);

    return `<div class="arr-cell" style="animation-delay:${i*0.03}s">
      <div class="${cls}">${flags.join('')}${s.score}</div>
      <div class="cell-idx">[${i}]</div>
      <div class="cell-name">${esc(s.name.split(' ')[0])}</div>
    </div>`;
  }).join('');
}

function renderFreq(modeRes) {
  const chart = document.getElementById('freq-chart');

  // find non-zero buckets
  const nonzero = [];
  for (let s = 0; s <= 100; s++) { if (freq[s] > 0) nonzero.push(s); }

  if (!nonzero.length) {
    chart.innerHTML = '<div class="freq-empty">freq[0..100] — all zeros</div>';
    document.getElementById('freq-chip').textContent = 'freq[] all zeros';
    return;
  }

  const maxFreq = Math.max(...nonzero.map(s => freq[s]));
  const MAX_H   = 130;

  document.getElementById('freq-chip').textContent =
    `${nonzero.length} distinct score${nonzero.length!==1?'s':''}`;

  chart.innerHTML = nonzero.map(s => {
    const h      = Math.max(4, Math.round((freq[s] / maxFreq) * MAX_H));
    const isMode = modeRes && s === modeRes.val && modeRes.freq > 1;
    return `<div class="freq-bar-wrap">
      <div class="freq-count">${freq[s]}</div>
      <div class="freq-bar ${isMode?'mode':'normal'}" style="height:${h}px"></div>
      <div class="freq-label">${s}</div>
    </div>`;
  }).join('');
}

function renderList(minVal, maxVal, modeRes) {
  const list = document.getElementById('score-list');
  if (!scores.length) {
    list.innerHTML = '<div class="list-empty">No student records</div>';
    document.getElementById('avg-chip').textContent = 'avg: —';
    return;
  }

  const avg = (scores.reduce((a,b)=>a+b.score,0)/scores.length).toFixed(1);
  document.getElementById('avg-chip').textContent = `avg: ${avg}`;

  list.innerHTML = scores.map((s, i) => {
    const isMin  = s.score === minVal;
    const isMax  = s.score === maxVal;
    const isMode = modeRes && s.score === modeRes.val && modeRes.freq > 1;
    let rowCls = 'score-row';
    if (isMode) rowCls += ' row-mode';
    else if (isMin) rowCls += ' row-min';
    else if (isMax) rowCls += ' row-max';

    let scoreCls = 'sr-score mid';
    if (s.score >= 80) scoreCls = 'sr-score high';
    if (s.score < 50)  scoreCls = 'sr-score low';

    return `<div class="${rowCls}">
      <span class="sr-num">${i+1}</span>
      <span class="sr-name">${esc(s.name)}</span>
      <span class="${scoreCls}">${s.score}</span>
      <button class="sr-del" onclick="removeScore(${i})" title="Remove">✕</button>
    </div>`;
  }).join('');
}

function renderCounts() {
  document.getElementById('count-chip').textContent = `${scores.length} entr${scores.length===1?'y':'ies'}`;
}

// ── Helpers ──────────────────────────────────────────────────
function esc(s) { return String(s).replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;'); }

function log(msg, type='') {
  const body = document.getElementById('log-body');
  const div  = document.createElement('div');
  div.className = 'log-entry';
  div.innerHTML = `<span class="log-t">${timeNow()}</span><span class="log-m ${type}">${msg}</span>`;
  body.prepend(div);
  while (body.children.length > 40) body.lastChild.remove();
}

function timeNow() {
  return new Date().toLocaleTimeString('en-IN',{hour:'2-digit',minute:'2-digit',second:'2-digit'});
}

let toastT;
function toast(msg, err=false) {
  const el = document.getElementById('toast');
  el.textContent = msg;
  el.className = 'show' + (err?' err':'');
  clearTimeout(toastT);
  toastT = setTimeout(()=>el.className='', 2300);
}
